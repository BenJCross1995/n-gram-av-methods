#!/bin/bash --login

#SBATCH -J combine_raw_scores
#SBATCH -t 4-0
#SBATCH -p gpuL
#SBATCH -G 1

set -e

# Load conda module
module load apps/binapps/anaconda3/2024.10

# Activate the environment
conda activate venv

BASE_LOC="/mnt/hum01-home01/d34195bc/av_datasets_experiments/concat_ngram_masking_logprobs"
SCRIPT_LOC="/mnt/hum01-home01/d34195bc/projects/n-gram-av-methods/scripts/combine_files.py"

# If any other models used add here
MODEL_LIST=(
    "gpt2"
)

# List of corpora - we will loop across all (skipping if data does not exist)
CORPORA=(
    "ACL"
    "Amazon"
    "Enron Cleaned"
    "Koppel's Blogs"
    "Perverted Justice"
    "Reddit"
    "StackExchange"
    "The Apricity"
    "The Telegraph"
    "TripAdvisor"
    "Wiki"
    "Yelp"
)

CORPORA=(
    "ACL"
)

DATA_TYPES=(
    "test"
    "training"
)

DATA_TYPES=(
    "test"
)

# raw/raw_10/raw_20/etc folders inside each model directory
RAW_DIRS=(
    "raw"
    "raw_10"
    "raw_20"
    "raw_30"
    "raw_40"
    "raw_50"
    "raw_60"
    "raw_70"
    "raw_80"
    "raw_90"
    "raw_100"
    "raw_200"
    "raw_300"
    "raw_400"
    "raw_500"
    "raw_600"
    "raw_700"
    "raw_800"
    "raw_900"
    "raw_1000"
)

# Excel sheet(s) to combine in step 1
SHEET_NAMES=(
    "metadata"
)

# Keep only completed == True if column exists
COMPLETE_ONLY=true

############################################
# OVERWRITE SETTINGS
############################################
# STEP 1: combine xlsx files within raw/raw_10/etc folders
OVERWRITE_STEP1=false
# STEP 2: combine raw-level RDS files to model-level
OVERWRITE_STEP2=true
# STEP 3: combine model-level RDS files to corpus-level
OVERWRITE_STEP3=true
# STEP 4: combine corpus-level RDS files to data_type-level
OVERWRITE_STEP4=true

############################################
# STEP 1: combine xlsx files within raw/raw_10/etc folders
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do
        for MODEL in "${MODEL_LIST[@]}"; do

            MODEL_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}"
            RAW_RESULTS_DIR="${MODEL_LOC}/raw_results"

            if [ ! -d "${MODEL_LOC}" ]; then
                echo "Skipping missing model directory: ${MODEL_LOC}"
                continue
            fi

            mkdir -p "${RAW_RESULTS_DIR}"

            for RAW_DIR in "${RAW_DIRS[@]}"; do

                DATA_LOC="${MODEL_LOC}/${RAW_DIR}"
                SAVE_LOC="${RAW_RESULTS_DIR}/${RAW_DIR}_complete_results.rds"

                if [ ! -d "${DATA_LOC}" ]; then
                    echo "Skipping missing directory: ${DATA_LOC}"
                    continue
                fi

                CMD=(
                    python -u "${SCRIPT_LOC}"
                    --mode combine_xlsx
                    --combine_level raw_subdir
                    --data_loc "${DATA_LOC}"
                    --save_loc "${SAVE_LOC}"
                    --sheet_names "${SHEET_NAMES[@]}"
                )

                if [ "${OVERWRITE_STEP1}" = true ]; then
                    CMD+=(--overwrite)
                fi

                if [ "${COMPLETE_ONLY}" = true ]; then
                    CMD+=(--complete_only)
                fi

                "${CMD[@]}"
            done
        done
    done
done

############################################
# STEP 2: combine raw-level RDS files to model-level
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do
        for MODEL in "${MODEL_LIST[@]}"; do

            MODEL_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}"
            RAW_RESULTS_DIR="${MODEL_LOC}/raw_results"
            SAVE_LOC="${MODEL_LOC}/combined_complete_results_raw.rds"

            if [ ! -d "${RAW_RESULTS_DIR}" ]; then
                echo "Skipping missing raw_results directory: ${RAW_RESULTS_DIR}"
                continue
            fi

            INPUT_FILES=()
            for RAW_DIR in "${RAW_DIRS[@]}"; do
                FILE="${RAW_RESULTS_DIR}/${RAW_DIR}_complete_results.rds"
                if [ -f "${FILE}" ]; then
                    INPUT_FILES+=("${FILE}")
                else
                    echo "Missing raw-level file, skipping: ${FILE}"
                fi
            done

            if [ "${#INPUT_FILES[@]}" -eq 0 ]; then
                echo "No raw-level files found for model: ${MODEL_LOC}"
                continue
            fi

            CMD=(
                python -u "${SCRIPT_LOC}"
                --mode combine_rds
                --combine_level model
                --save_loc "${SAVE_LOC}"
                --input_files "${INPUT_FILES[@]}"
            )

            if [ "${OVERWRITE_STEP2}" = true ]; then
                CMD+=(--overwrite)
            fi

            if [ "${COMPLETE_ONLY}" = true ]; then
                CMD+=(--complete_only)
            fi

            "${CMD[@]}"
        done
    done
done

############################################
# STEP 3: combine model-level RDS files to corpus-level
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do

        CORPUS_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}"
        SAVE_LOC="${CORPUS_LOC}/combined_complete_results_raw.rds"

        if [ ! -d "${CORPUS_LOC}" ]; then
            echo "Skipping missing corpus directory: ${CORPUS_LOC}"
            continue
        fi

        INPUT_FILES=()
        for MODEL in "${MODEL_LIST[@]}"; do
            FILE="${CORPUS_LOC}/${MODEL}/combined_complete_results_raw.rds"
            if [ -f "${FILE}" ]; then
                INPUT_FILES+=("${FILE}")
            else
                echo "Missing model-level file, skipping: ${FILE}"
            fi
        done

        if [ "${#INPUT_FILES[@]}" -eq 0 ]; then
            echo "No model-level files found for corpus: ${CORPUS_LOC}"
            continue
        fi

        CMD=(
            python -u "${SCRIPT_LOC}"
            --mode combine_rds
            --combine_level corpus
            --save_loc "${SAVE_LOC}"
            --input_files "${INPUT_FILES[@]}"
        )

        if [ "${OVERWRITE_STEP3}" = true ]; then
            CMD+=(--overwrite)
        fi

        if [ "${COMPLETE_ONLY}" = true ]; then
            CMD+=(--complete_only)
        fi

        "${CMD[@]}"

    done
done

############################################
# STEP 4: combine corpus-level RDS files to data_type-level
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do

    DATA_TYPE_LOC="${BASE_LOC}/${DATA_TYPE}"
    SAVE_LOC="${DATA_TYPE_LOC}/combined_complete_results_raw.rds"

    if [ ! -d "${DATA_TYPE_LOC}" ]; then
        echo "Skipping missing data_type directory: ${DATA_TYPE_LOC}"
        continue
    fi

    INPUT_FILES=()
    for CORPUS in "${CORPORA[@]}"; do
        FILE="${DATA_TYPE_LOC}/${CORPUS}/combined_complete_results_raw.rds"
        if [ -f "${FILE}" ]; then
            INPUT_FILES+=("${FILE}")
        else
            echo "Missing corpus-level file, skipping: ${FILE}"
        fi
    done

    if [ "${#INPUT_FILES[@]}" -eq 0 ]; then
        echo "No corpus-level files found for data type: ${DATA_TYPE_LOC}"
        continue
    fi

    CMD=(
        python -u "${SCRIPT_LOC}"
        --mode combine_rds
        --combine_level data_type
        --save_loc "${SAVE_LOC}"
        --input_files "${INPUT_FILES[@]}"
    )

    if [ "${OVERWRITE_STEP4}" = true ]; then
        CMD+=(--overwrite)
    fi

    if [ "${COMPLETE_ONLY}" = true ]; then
        CMD+=(--complete_only)
    fi

    "${CMD[@]}"

done

echo "Finished successfully."
