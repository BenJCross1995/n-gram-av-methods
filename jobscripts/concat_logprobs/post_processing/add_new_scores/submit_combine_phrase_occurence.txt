# ============================================================

sanitize_name() {
    echo "$1" | tr -cs '[:alnum:]' '_'
}

submit_wrap_job() {
    local job_name="$1"
    local dependency="$2"
    local script_body="$3"

    local cmd=(
        sbatch
        --parsable
        -J "${job_name}"
        -p himem
        -t 7-0
        -n 1
        --mem=64G
        -o "${LOG_DIR}/%x-%A.out"
        -e "${LOG_DIR}/%x-%A.err"
        --export=ALL
    )

    if [[ -n "${dependency}" ]]; then
        cmd+=(--dependency="${dependency}")
    fi

    local escaped_body
    escaped_body=$(printf '%q' "${script_body}")

    cmd+=(--wrap "bash --login -c ${escaped_body}")

    "${cmd[@]}"
}

build_pair_combine_body() {
    local data_type="$1"
    local corpus="$2"

    cat <<EOF
#!/bin/bash --login
set -euo pipefail

module purge
module load "${CONDA_MODULE}"
conda activate "${CONDA_ENV}"

IFS='|' read -r -a MODEL_LIST <<< "\${MODEL_LIST_STR}"
IFS='|' read -r -a RAW_DIRS <<< "\${RAW_DIRS_STR}"

run_combine_rds() {
    local combine_level="\$1"
    local save_loc="\$2"
    local overwrite_flag="\$3"
    shift 3
    local input_files=( "\$@" )

    if [[ "\${#input_files[@]}" -eq 0 ]]; then
        echo "No input files for \${combine_level} -> \${save_loc}; skipping"
        return 0
    fi

    local cmd=(
        python -u "\${PYTHON_COMBINE_SCRIPT}"
        --mode combine_rds
        --combine_level "\${combine_level}"
        --save_loc "\${save_loc}"
        --input_files "\${input_files[@]}"
    )

    if [[ "\${overwrite_flag}" == "true" ]]; then
        cmd+=(--overwrite)
    fi

    if [[ "\${COMPLETE_ONLY}" == "true" ]]; then
        cmd+=(--complete_only)
    fi

    echo "Combining level=\${combine_level}"
    echo "  SAVE_LOC=\${save_loc}"
    printf '  INPUT=%s\n' "\${input_files[@]}"

    "\${cmd[@]}"
}

run_postprocess_model_file() {
    local data_loc="\$1"
    local save_loc="\$2"
    local overwrite_flag="\$3"

    if [[ ! -f "\${data_loc}" ]]; then
        echo "Skipping missing postprocess input: \${data_loc}"
        return 0
    fi

    local cmd=(
        python -u "\${POSTPROCESS_SCRIPT}"
        --data_loc "\${data_loc}"
        --save_loc "\${save_loc}"
    )

    if [[ "\${overwrite_flag}" == "true" ]]; then
        cmd+=(--overwrite)
    fi

    if [[ "\${POSTPROCESS_INCLUDE_SUMS}" == "true" ]]; then
        cmd+=(--include_sums)
    fi

    echo "Running model-level postprocess"
    echo "  DATA_LOC=\${data_loc}"
    echo "  SAVE_LOC=\${save_loc}"

    "\${cmd[@]}"
}

DATA_TYPE="${data_type}"
CORPUS="${corpus}"

echo "Starting pair-level combine for DATA_TYPE=\${DATA_TYPE}, CORPUS=\${CORPUS}"

for MODEL in "\${MODEL_LIST[@]}"; do
    MODEL_LOC="\${BASE_LOC}/\${DATA_TYPE}/\${CORPUS}/\${MODEL}"
    RAW_RESULTS_DIR="\${MODEL_LOC}/raw_results"

    MODEL_COMBINED_SAVE_LOC="\${MODEL_LOC}/_results/combined_\${OUTPUT_STEM}.rds"
    MODEL_POSTPROCESS_SAVE_LOC="\${MODEL_LOC}/_results/combined_\${POSTPROCESS_OUTPUT_STEM}.rds"

    mkdir -p "\${MODEL_LOC}/_results"

    if [[ ! -d "\${RAW_RESULTS_DIR}" ]]; then
        echo "Skipping missing raw_results directory: \${RAW_RESULTS_DIR}"
        continue
    fi

    INPUT_FILES=()
    for RAW_DIR in "\${RAW_DIRS[@]}"; do
        FILE="\${RAW_RESULTS_DIR}/\${RAW_DIR}_\${OUTPUT_STEM}.rds"
        [[ -f "\${FILE}" ]] && INPUT_FILES+=("\${FILE}")
    done

    # Step 2: model-level combine of original occurrence-level files
    run_combine_rds model "\${MODEL_COMBINED_SAVE_LOC}" "\${STEP2_OVERWRITE}" "\${INPUT_FILES[@]}"

    # Step 2b: derive the summary-by-token file at model level
    run_postprocess_model_file "\${MODEL_COMBINED_SAVE_LOC}" "\${MODEL_POSTPROCESS_SAVE_LOC}" "\${STEP2B_OVERWRITE}"
done

# Step 3: corpus-level combine now uses the NEW derived model files
CORPUS_LOC="\${BASE_LOC}/\${DATA_TYPE}/\${CORPUS}"
SAVE_LOC="\${CORPUS_LOC}/_results/combined_\${POSTPROCESS_OUTPUT_STEM}.rds"

mkdir -p "\${CORPUS_LOC}/_results"

INPUT_FILES=()
for MODEL in "\${MODEL_LIST[@]}"; do
    FILE="\${CORPUS_LOC}/\${MODEL}/_results/combined_\${POSTPROCESS_OUTPUT_STEM}.rds"
    [[ -f "\${FILE}" ]] && INPUT_FILES+=("\${FILE}")
done

run_combine_rds corpus "\${SAVE_LOC}" "\${STEP3_OVERWRITE}" "\${INPUT_FILES[@]}"

echo "Finished pair-level combine for DATA_TYPE=\${DATA_TYPE}, CORPUS=\${CORPUS}"
EOF
}

#!/bin/bash --login
set -euo pipefail

module purge

IFS='|' read -r -a DATA_TYPES <<< "\${DATA_TYPES_STR}"

run_combine_rds() {

# ============================================================
# Submit pipeline
# ============================================================
echo "Submitting phrase occurrence score combine pipeline"
echo "STEP1/STEP2 OUTPUT_STEM=${OUTPUT_STEM}"
echo "DOWNSTREAM POSTPROCESS_OUTPUT_STEM=${POSTPROCESS_OUTPUT_STEM}"
echo "SHEETS=${SHEET_NAMES_STR}"

declare -A CORPUS_JOB_IDS_BY_DATATYPE
OVERALL_PAIR_JOB_IDS=()

for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do
        safe_pair="$(sanitize_name "${DATA_TYPE}__${CORPUS}")"
        TASK_FILE="${TASK_DIR}/combine_files_array_${safe_pair}.txt"
        : > "${TASK_FILE}"

        task_count=0
        for MODEL in "${MODEL_LIST[@]}"; do
            for RAW_DIR in "${RAW_DIRS[@]}"; do
                printf '%s\t%s\t%s\t%s\n' \
                    "${DATA_TYPE}" \
                    "${CORPUS}" \
                    "${MODEL}" \
                    "${RAW_DIR}" >> "${TASK_FILE}"
                task_count=$((task_count + 1))
            done
        done

        if [[ "${task_count}" -eq 0 ]]; then
            echo "No tasks for ${DATA_TYPE} / ${CORPUS}; skipping"
            continue
        fi

        echo
        echo "Submitting array job for ${DATA_TYPE} / ${CORPUS}"
        echo "  TASK_FILE=${TASK_FILE}"
        echo "  TASKS=${task_count}"

        array_job_id="$(
            sbatch --parsable \
                -J "combine_phrase_${safe_pair}" \
                --array="1-${task_count}%${COMBINE_ARRAY_MAX_CONCURRENT}" \
                --export=ALL,TASK_FILE="${TASK_FILE}" \
                "${ARRAY_SCRIPT}"
        )"

        echo "  Array job id: ${array_job_id}"

        pair_body="$(build_pair_combine_body "${DATA_TYPE}" "${CORPUS}")"
        pair_job_id="$(
            submit_wrap_job \
                "combine_phrase_post_${safe_pair}" \
                "afterok:${array_job_id}" \
                "${pair_body}"
        )"

        echo "  Pair combine job id: ${pair_job_id}"

        OVERALL_PAIR_JOB_IDS+=("${pair_job_id}")

        if [[ -z "${CORPUS_JOB_IDS_BY_DATATYPE[${DATA_TYPE}]:-}" ]]; then
            CORPUS_JOB_IDS_BY_DATATYPE["${DATA_TYPE}"]="${pair_job_id}"
        else
            CORPUS_JOB_IDS_BY_DATATYPE["${DATA_TYPE}"]="${CORPUS_JOB_IDS_BY_DATATYPE[${DATA_TYPE}]}:${pair_job_id}"
        fi
    done
done

DATA_TYPE_JOB_IDS=()

for DATA_TYPE in "${DATA_TYPES[@]}"; do
    dep_ids="${CORPUS_JOB_IDS_BY_DATATYPE[${DATA_TYPE}]:-}"

    if [[ -z "${dep_ids}" ]]; then
        echo "No corpus-level jobs recorded for ${DATA_TYPE}; skipping data_type submit"
        continue
    fi

    safe_dt="$(sanitize_name "${DATA_TYPE}")"
    dt_body="$(build_datatype_combine_body "${DATA_TYPE}")"

    dt_job_id="$(
        submit_wrap_job \
            "combine_phrase_dtype_${safe_dt}" \
            "afterok:${dep_ids}" \
            "${dt_body}"
    )"

    echo "Submitted data_type combine for ${DATA_TYPE}: ${dt_job_id}"
    DATA_TYPE_JOB_IDS+=("${dt_job_id}")
done

if [[ "${#DATA_TYPE_JOB_IDS[@]}" -gt 0 ]]; then
    overall_dep="afterok:$(IFS=:; echo "${DATA_TYPE_JOB_IDS[*]}")"
    overall_body="$(build_overall_combine_body)"

    overall_job_id="$(
        submit_wrap_job \
            "combine_phrase_overall" \
            "${overall_dep}" \
            "${overall_body}"
    )"

    echo
    echo "Submitted overall combine job: ${overall_job_id}"
else
    echo "No data_type jobs were submitted, so no overall combine job was submitted."
fi

echo
echo "Submission complete."
