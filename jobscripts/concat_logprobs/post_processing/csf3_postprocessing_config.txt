clean_conda_from_path() {
    conda deactivate 2>/dev/null || true

    unset CPATH C_INCLUDE_PATH CPLUS_INCLUDE_PATH LIBRARY_PATH

    hash -r
}

require_task_file() {
    if [[ -z "${TASK_FILE:-}" ]]; then
        echo "TASK_FILE environment variable is not set."
        exit 1
    fi

    if [[ ! -f "${TASK_FILE}" ]]; then
        echo "TASK_FILE does not exist: ${TASK_FILE}"
        exit 1
    fi
}

performance_prefix_for_input() {
    local input_file
    input_file="$(basename "$1")"

    case "${input_file}" in
        combined_complete_results_raw.rds)
            echo "performance"
            ;;
        combined_adjusted_scores_raw.rds)
            echo "performance_adj"
            ;;
        combined_like_for_like_scores_raw.rds)
            echo "lfl_performance"
            ;;
        combined_like_for_like_adjusted_scores_raw.rds)
            echo "lfl_performance_adj"
            ;;
        *)
            local stem
            stem="${input_file%.rds}"
            stem="${stem#combined_}"
            stem="${stem%_raw}"
            stem="${stem//_scores/}"
            stem="${stem//_results/}"
            stem="${stem//__/_}"
            stem="$(echo "${stem}" | sed 's/^_//; s/_$//')"

            if [[ -z "${stem}" ]]; then
                echo "performance"
            else
                echo "performance_${stem}"
            fi
            ;;
    esac
}
