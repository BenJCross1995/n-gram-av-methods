#!/bin/bash --login

#SBATCH -J lfl_post
#SBATCH -p himem
#SBATCH -t 7-0
#SBATCH -n 1
#SBATCH --mem=64G
#SBATCH -o /mnt/hum01-home01/d34195bc/projects/n-gram-av-methods/slurm_logs/postprocessing/%x-%A.out
#SBATCH -e /mnt/hum01-home01/d34195bc/projects/n-gram-av-methods/slurm_logs/postprocessing/%x-%A.err

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
source "${SCRIPT_DIR}/csf3_postprocessing_config.txt"

load_python_env

run_combine_rds() {
    local combine_level="$1"
    local save_loc="$2"
    local overwrite_flag="$3"
    shift 3
    local input_files=("$@")

    if [[ "${#input_files[@]}" -eq 0 ]]; then
        echo "No input files for ${combine_level} -> ${save_loc}; skipping"
        return 0
    fi

    local cmd=(
        python -u "${PYTHON_COMBINE_SCRIPT}"
        --mode combine_rds
        --combine_level "${combine_level}"
        --save_loc "${save_loc}"
        --input_files "${input_files[@]}"
    )

    if [[ "${overwrite_flag}" == true ]]; then
        cmd+=(--overwrite)
    fi

    echo "Combining level=${combine_level}"
    echo "  SAVE_LOC=${save_loc}"
    printf '  INPUT=%s\n' "${input_files[@]}"

    "${cmd[@]}"
}

OUTPUT_FILE_NAMES=(
    "combined_adjusted_scores_raw.rds"
    "combined_like_for_like_scores_raw.rds"
    "combined_like_for_like_adjusted_scores_raw.rds"
)

echo "Starting like-for-like post-processing (steps 2-4)"

# --------------------------------------------------
# Step 2: model-level outputs -> corpus-level outputs
# --------------------------------------------------
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do
        CORPUS_RESULTS_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/_results"
        mkdir -p "${CORPUS_RESULTS_LOC}"

        for OUTPUT_FILE in "${OUTPUT_FILE_NAMES[@]}"; do
            SAVE_LOC="${CORPUS_RESULTS_LOC}/${OUTPUT_FILE}"

            INPUT_FILES=()
            for MODEL in "${MODEL_LIST[@]}"; do
                FILE="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/_results/${OUTPUT_FILE}"
                [[ -f "${FILE}" ]] && INPUT_FILES+=("${FILE}")
            done

            run_combine_rds corpus "${SAVE_LOC}" "${LFL_OVERWRITE_CORPUS_LEVEL}" "${INPUT_FILES[@]}"
        done
    done
done

# --------------------------------------------------
# Step 3: corpus-level outputs -> data_type-level outputs
# --------------------------------------------------
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    DATA_TYPE_RESULTS_LOC="${BASE_LOC}/${DATA_TYPE}/_results"
    mkdir -p "${DATA_TYPE_RESULTS_LOC}"

    for OUTPUT_FILE in "${OUTPUT_FILE_NAMES[@]}"; do
        SAVE_LOC="${DATA_TYPE_RESULTS_LOC}/${OUTPUT_FILE}"

        INPUT_FILES=()
        for CORPUS in "${CORPORA[@]}"; do
            FILE="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/_results/${OUTPUT_FILE}"
            [[ -f "${FILE}" ]] && INPUT_FILES+=("${FILE}")
        done

        run_combine_rds data_type "${SAVE_LOC}" "${LFL_OVERWRITE_DATA_TYPE_LEVEL}" "${INPUT_FILES[@]}"
    done
done

# --------------------------------------------------
# Step 4: data_type-level outputs -> overall-level outputs
# --------------------------------------------------
OVERALL_RESULTS_LOC="${BASE_LOC}/_results"
mkdir -p "${OVERALL_RESULTS_LOC}"

for OUTPUT_FILE in "${OUTPUT_FILE_NAMES[@]}"; do
    SAVE_LOC="${OVERALL_RESULTS_LOC}/${OUTPUT_FILE}"

    INPUT_FILES=()
    for DATA_TYPE in "${DATA_TYPES[@]}"; do
        FILE="${BASE_LOC}/${DATA_TYPE}/_results/${OUTPUT_FILE}"
        [[ -f "${FILE}" ]] && INPUT_FILES+=("${FILE}")
    done

    run_combine_rds overall "${SAVE_LOC}" "${LFL_OVERWRITE_OVERALL_LEVEL}" "${INPUT_FILES[@]}"
done

echo "Finished like-for-like post-processing."
