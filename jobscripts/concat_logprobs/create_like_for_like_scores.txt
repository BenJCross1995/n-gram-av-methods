#!/bin/bash --login

#SBATCH -J threshold_versions_split
#SBATCH -t 4-0
#SBATCH -p gpuL
#SBATCH -G 1

set -euo pipefail

# Load conda module
module load apps/binapps/anaconda3/2024.10

# Activate the environment
conda activate venv

BASE_LOC="/mnt/hum01-home01/d34195bc/av_datasets_experiments/concat_ngram_masking_logprobs"
ADJUST_SCRIPT_LOC="/mnt/hum01-home01/d34195bc/projects/n-gram-av-methods/scripts/create_like_for_like.py"
COMBINE_SCRIPT_LOC="/mnt/hum01-home01/d34195bc/projects/n-gram-av-methods/scripts/combine_files.py"

MODEL_LIST=(
    "gpt2"
)

CORPORA=(
    "ACL"
    "Amazon"
    "Enron Cleaned"
    "Koppel's Blogs"
    "Perverted Justice"
    "Reddit"
    "StackExchange"
    "The Apricity"
    "The Telegraph"
    "TripAdvisor"
    "Wiki"
    "Yelp"
)

DATA_TYPES=(
    "test"
    "training"
)

############################################
# FILE NAMES
############################################
INPUT_FILE_NAME="combined_complete_results_raw.rds"

ADJUSTED_SAVE_NAME="combined_adjusted_scores_raw.rds"
LIKE_FOR_LIKE_SAVE_NAME="combined_like_for_like_scores_raw.rds"
LIKE_FOR_LIKE_ADJUSTED_SAVE_NAME="combined_like_for_like_adjusted_scores_raw.rds"

OUTPUT_FILE_NAMES=(
    "${ADJUSTED_SAVE_NAME}"
    "${LIKE_FOR_LIKE_SAVE_NAME}"
    "${LIKE_FOR_LIKE_ADJUSTED_SAVE_NAME}"
)

############################################
# OVERWRITE SETTINGS
############################################
# Step 1: run adjustment only at model level
OVERWRITE_MODEL_LEVEL=true

# Step 2: combine model -> corpus
OVERWRITE_CORPUS_LEVEL=true

# Step 3: combine corpus -> data_type
OVERWRITE_DATA_TYPE_LEVEL=true

# Step 4: combine data_type -> overall
OVERWRITE_OVERALL_LEVEL=true

############################################
# STEP 1: create adjusted outputs at model level only
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do
        for MODEL in "${MODEL_LIST[@]}"; do

            MODEL_RESULTS_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/_results"

            INPUT_LOC="${MODEL_RESULTS_LOC}/${INPUT_FILE_NAME}"
            ADJUSTED_SAVE_LOC="${MODEL_RESULTS_LOC}/${ADJUSTED_SAVE_NAME}"
            LIKE_FOR_LIKE_SAVE_LOC="${MODEL_RESULTS_LOC}/${LIKE_FOR_LIKE_SAVE_NAME}"
            LIKE_FOR_LIKE_ADJUSTED_SAVE_LOC="${MODEL_RESULTS_LOC}/${LIKE_FOR_LIKE_ADJUSTED_SAVE_NAME}"

            if [ ! -f "${INPUT_LOC}" ]; then
                echo "Skipping missing model-level input: ${INPUT_LOC}"
                continue
            fi

            CMD=(
                python -u "${ADJUST_SCRIPT_LOC}"
                --input_loc "${INPUT_LOC}"
                --adjusted_save_loc "${ADJUSTED_SAVE_LOC}"
                --like_for_like_save_loc "${LIKE_FOR_LIKE_SAVE_LOC}"
                --like_for_like_adjusted_save_loc "${LIKE_FOR_LIKE_ADJUSTED_SAVE_LOC}"
            )

            if [ "${OVERWRITE_MODEL_LEVEL}" = true ]; then
                CMD+=(--overwrite)
            fi

            echo "Running model-level adjustment:"
            echo "  DATA_TYPE=${DATA_TYPE}"
            echo "  CORPUS=${CORPUS}"
            echo "  MODEL=${MODEL}"
            "${CMD[@]}"
        done
    done
done

############################################
# STEP 2: combine model-level outputs to corpus-level
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do

        CORPUS_RESULTS_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/_results"
        mkdir -p "${CORPUS_RESULTS_LOC}"

        for OUTPUT_FILE in "${OUTPUT_FILE_NAMES[@]}"; do

            SAVE_LOC="${CORPUS_RESULTS_LOC}/${OUTPUT_FILE}"

            INPUT_FILES=()
            for MODEL in "${MODEL_LIST[@]}"; do
                FILE="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/_results/${OUTPUT_FILE}"
                if [ -f "${FILE}" ]; then
                    INPUT_FILES+=("${FILE}")
                else
                    echo "Missing model-level file, skipping: ${FILE}"
                fi
            done

            if [ "${#INPUT_FILES[@]}" -eq 0 ]; then
                echo "No model-level files found for corpus combine: ${DATA_TYPE} / ${CORPUS} / ${OUTPUT_FILE}"
                continue
            fi

            CMD=(
                python -u "${COMBINE_SCRIPT_LOC}"
                --mode combine_rds
                --combine_level corpus
                --save_loc "${SAVE_LOC}"
                --input_files "${INPUT_FILES[@]}"
            )

            if [ "${OVERWRITE_CORPUS_LEVEL}" = true ]; then
                CMD+=(--overwrite)
            fi

            echo "Combining model -> corpus:"
            echo "  DATA_TYPE=${DATA_TYPE}"
            echo "  CORPUS=${CORPUS}"
            echo "  OUTPUT=${OUTPUT_FILE}"
            "${CMD[@]}"
        done
    done
done

############################################
# STEP 3: combine corpus-level outputs to data_type-level
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do

    DATA_TYPE_RESULTS_LOC="${BASE_LOC}/${DATA_TYPE}/_results"
    mkdir -p "${DATA_TYPE_RESULTS_LOC}"

    for OUTPUT_FILE in "${OUTPUT_FILE_NAMES[@]}"; do

        SAVE_LOC="${DATA_TYPE_RESULTS_LOC}/${OUTPUT_FILE}"

        INPUT_FILES=()
        for CORPUS in "${CORPORA[@]}"; do
            FILE="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/_results/${OUTPUT_FILE}"
            if [ -f "${FILE}" ]; then
                INPUT_FILES+=("${FILE}")
            else
                echo "Missing corpus-level file, skipping: ${FILE}"
            fi
        done

        if [ "${#INPUT_FILES[@]}" -eq 0 ]; then
            echo "No corpus-level files found for data_type combine: ${DATA_TYPE} / ${OUTPUT_FILE}"
            continue
        fi

        CMD=(
            python -u "${COMBINE_SCRIPT_LOC}"
            --mode combine_rds
            --combine_level data_type
            --save_loc "${SAVE_LOC}"
            --input_files "${INPUT_FILES[@]}"
        )

        if [ "${OVERWRITE_DATA_TYPE_LEVEL}" = true ]; then
            CMD+=(--overwrite)
        fi

        echo "Combining corpus -> data_type:"
        echo "  DATA_TYPE=${DATA_TYPE}"
        echo "  OUTPUT=${OUTPUT_FILE}"
        "${CMD[@]}"
    done
done

############################################
# STEP 4: combine data_type-level outputs to overall-level
############################################
OVERALL_RESULTS_LOC="${BASE_LOC}/_results"
mkdir -p "${OVERALL_RESULTS_LOC}"

for OUTPUT_FILE in "${OUTPUT_FILE_NAMES[@]}"; do

    SAVE_LOC="${OVERALL_RESULTS_LOC}/${OUTPUT_FILE}"

    INPUT_FILES=()
    for DATA_TYPE in "${DATA_TYPES[@]}"; do
        FILE="${BASE_LOC}/${DATA_TYPE}/_results/${OUTPUT_FILE}"
        if [ -f "${FILE}" ]; then
            INPUT_FILES+=("${FILE}")
        else
            echo "Missing data_type-level file, skipping: ${FILE}"
        fi
    done

    if [ "${#INPUT_FILES[@]}" -eq 0 ]; then
        echo "No data_type-level files found for overall combine: ${OUTPUT_FILE}"
        continue
    fi

    CMD=(
        python -u "${COMBINE_SCRIPT_LOC}"
        --mode combine_rds
        --combine_level overall
        --save_loc "${SAVE_LOC}"
        --input_files "${INPUT_FILES[@]}"
    )

    if [ "${OVERWRITE_OVERALL_LEVEL}" = true ]; then
        CMD+=(--overwrite)
    fi

    echo "Combining data_type -> overall:"
    echo "  OUTPUT=${OUTPUT_FILE}"
    "${CMD[@]}"
done

echo "Finished successfully."
