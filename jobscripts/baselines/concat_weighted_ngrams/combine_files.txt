#!/bin/bash --login

#SBATCH -J combine_scores
#SBATCH -p himem
#SBATCH -t 7-0
#SBATCH -n 1
#SBATCH --mem=64G

set -euo pipefail
shopt -s nullglob

# Load conda module
module load apps/binapps/anaconda3/2024.10

# Activate the environment
conda activate venv

BASE_LOC="/mnt/hum01-home01/d34195bc/av_datasets_experiments/weighted_ngram_tracing_concat_v2"
SCRIPT_LOC="/mnt/hum01-home01/d34195bc/projects/n-gram-av-methods/scripts/weighted_ngram_tracing_combine_files.py"

MODEL_LIST=(
    "gpt2"
)

CORPORA=(
    "ACL"
    "All-the-news"
    "Amazon"
    "Enron Cleaned"
    "IMDB"
    "Koppel's Blogs"
    "Perverted Justice"
    "Reddit"
    "StackExchange"
    "The Apricity"
    "The Telegraph"
    "TripAdvisor"
    "Wiki"
    "Yelp"
)

DATA_TYPES=(
    "test"
    "training"
)

WEIGHTS=(
    "none"
    "linear"
    "linear_rough"
    "power"
    "exp"
)

############################################
# OVERWRITE SETTINGS
############################################
OVERWRITE_STEP1=false   # leaf xlsx -> leaf/_results
OVERWRITE_STEP2=true    # weight-level _results
OVERWRITE_STEP3=true    # model-level _results
OVERWRITE_STEP4=true    # corpus-level _results
OVERWRITE_STEP5=true    # data_type-level _results
OVERWRITE_STEP6=true    # overall _results

############################################
# HELPERS
############################################

run_combine_xlsx() {
    local DATA_LOC="$1"
    local SAVE_LOC="$2"
    local OVERWRITE="$3"

    mkdir -p "$(dirname "${SAVE_LOC}")"

    CMD=(
        python -u "${SCRIPT_LOC}"
        --mode combine_xlsx
        --combine_level weight
        --data_loc "${DATA_LOC}"
        --save_loc "${SAVE_LOC}"
    )

    if [ "${OVERWRITE}" = true ]; then
        CMD+=(--overwrite)
    fi

    echo "Combining xlsx from: ${DATA_LOC}"
    echo "Saving to:           ${SAVE_LOC}"
    "${CMD[@]}"
}

run_combine_rds() {
    local COMBINE_LEVEL="$1"
    local SAVE_LOC="$2"
    local OVERWRITE="$3"
    shift 3
    local INPUT_FILES=("$@")

    if [ "${#INPUT_FILES[@]}" -eq 0 ]; then
        echo "Skipping ${COMBINE_LEVEL}: no input files"
        return 0
    fi

    mkdir -p "$(dirname "${SAVE_LOC}")"

    CMD=(
        python -u "${SCRIPT_LOC}"
        --mode combine_rds
        --combine_level "${COMBINE_LEVEL}"
        --save_loc "${SAVE_LOC}"
        --input_files "${INPUT_FILES[@]}"
    )

    if [ "${OVERWRITE}" = true ]; then
        CMD+=(--overwrite)
    fi

    echo "Combining rds (${COMBINE_LEVEL})"
    echo "Saving to: ${SAVE_LOC}"
    printf 'Inputs:\n'
    printf '  %s\n' "${INPUT_FILES[@]}"
    "${CMD[@]}"
}

############################################
# STEP 1:
# combine xlsx files in each leaf directory
#
# Examples of leaf directories:
#   .../model/none
#   .../model/linear
#   .../model/power/alpha_2.0
#   .../model/exp/base_1.5
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do
        for MODEL in "${MODEL_LIST[@]}"; do
            for WEIGHT in "${WEIGHTS[@]}"; do

                WEIGHT_DIR="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/${WEIGHT}"

                if [ ! -d "${WEIGHT_DIR}" ]; then
                    echo "Skipping missing weight directory: ${WEIGHT_DIR}"
                    continue
                fi

                # Find immediate child directories, excluding _results
                SUBDIRS=()
                for d in "${WEIGHT_DIR}"/*/; do
                    [ -d "${d}" ] || continue
                    d="${d%/}"
                    [ "$(basename "${d}")" = "_results" ] && continue
                    SUBDIRS+=("${d}")
                done

                if [ "${#SUBDIRS[@]}" -gt 0 ]; then
                    # Nested case: combine each child leaf dir
                    for SUBDIR in "${SUBDIRS[@]}"; do
                        SAVE_LOC="${SUBDIR}/_results/combined_scores_raw.rds"
                        run_combine_xlsx "${SUBDIR}" "${SAVE_LOC}" "${OVERWRITE_STEP1}"
                    done
                else
                    # Flat case: combine the weight dir itself
                    SAVE_LOC="${WEIGHT_DIR}/_results/combined_scores_raw.rds"
                    run_combine_xlsx "${WEIGHT_DIR}" "${SAVE_LOC}" "${OVERWRITE_STEP1}"
                fi

            done
        done
    done
done

############################################
# STEP 2:
# combine leaf-level RDS files to weight-level _results
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do
        for MODEL in "${MODEL_LIST[@]}"; do
            for WEIGHT in "${WEIGHTS[@]}"; do

                WEIGHT_DIR="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/${WEIGHT}"

                if [ ! -d "${WEIGHT_DIR}" ]; then
                    echo "Skipping missing weight directory: ${WEIGHT_DIR}"
                    continue
                fi

                INPUT_FILES=()

                SUBDIRS=()
                for d in "${WEIGHT_DIR}"/*/; do
                    [ -d "${d}" ] || continue
                    d="${d%/}"
                    [ "$(basename "${d}")" = "_results" ] && continue
                    SUBDIRS+=("${d}")
                done

                if [ "${#SUBDIRS[@]}" -gt 0 ]; then
                    for SUBDIR in "${SUBDIRS[@]}"; do
                        FILE="${SUBDIR}/_results/combined_scores_raw.rds"
                        [ -f "${FILE}" ] && INPUT_FILES+=("${FILE}")
                    done
                else
                    FILE="${WEIGHT_DIR}/_results/combined_scores_raw.rds"
                    [ -f "${FILE}" ] && INPUT_FILES+=("${FILE}")
                fi

                SAVE_LOC="${WEIGHT_DIR}/_results/combined_scores_raw.rds"
                run_combine_rds "weight" "${SAVE_LOC}" "${OVERWRITE_STEP2}" "${INPUT_FILES[@]}"

            done
        done
    done
done

############################################
# STEP 3:
# combine weight-level RDS files to model-level _results
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do
        for MODEL in "${MODEL_LIST[@]}"; do

            INPUT_FILES=()
            for WEIGHT in "${WEIGHTS[@]}"; do
                FILE="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/${WEIGHT}/_results/combined_scores_raw.rds"
                [ -f "${FILE}" ] && INPUT_FILES+=("${FILE}")
            done

            SAVE_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/_results/combined_scores_raw.rds"
            run_combine_rds "model" "${SAVE_LOC}" "${OVERWRITE_STEP3}" "${INPUT_FILES[@]}"

        done
    done
done

############################################
# STEP 4:
# combine model-level RDS files to corpus-level _results
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do

        INPUT_FILES=()
        for MODEL in "${MODEL_LIST[@]}"; do
            FILE="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/_results/combined_scores_raw.rds"
            [ -f "${FILE}" ] && INPUT_FILES+=("${FILE}")
        done

        SAVE_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/_results/combined_scores_raw.rds"
        run_combine_rds "corpus" "${SAVE_LOC}" "${OVERWRITE_STEP4}" "${INPUT_FILES[@]}"

    done
done

############################################
# STEP 5:
# combine corpus-level RDS files to data_type-level _results
############################################
for DATA_TYPE in "${DATA_TYPES[@]}"; do

    INPUT_FILES=()
    for CORPUS in "${CORPORA[@]}"; do
        FILE="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/_results/combined_scores_raw.rds"
        [ -f "${FILE}" ] && INPUT_FILES+=("${FILE}")
    done

    SAVE_LOC="${BASE_LOC}/${DATA_TYPE}/_results/combined_scores_raw.rds"
    run_combine_rds "data_type" "${SAVE_LOC}" "${OVERWRITE_STEP5}" "${INPUT_FILES[@]}"

done

############################################
# STEP 6:
# combine test + training to overall _results
############################################
INPUT_FILES=()

for DATA_TYPE in "${DATA_TYPES[@]}"; do
    FILE="${BASE_LOC}/${DATA_TYPE}/_results/combined_scores_raw.rds"
    [ -f "${FILE}" ] && INPUT_FILES+=("${FILE}")
done

SAVE_LOC="${BASE_LOC}/_results/combined_scores_raw.rds"
run_combine_rds "overall" "${SAVE_LOC}" "${OVERWRITE_STEP6}" "${INPUT_FILES[@]}"
