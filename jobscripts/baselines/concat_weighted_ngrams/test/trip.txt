#SBATCH -J trip_t
#SBATCH -a 1-480
#SBATCH -p himem
#SBATCH -t 7-0
#SBATCH -n 1
#SBATCH --mem=64G

set -e  # optional: stop on error

# Load conda module
module load apps/binapps/anaconda3/2024.10

# Activate the environment
conda activate venv

# Define variables
CORPUS="TripAdvisor"
DATA_TYPE="test"
MODEL="gpt2"
MODEL_LOC="/mnt/hum01-home01/d34195bc/models/${MODEL}"
BASE="2.0"
ALPHA="1.0"

# Input lists produced in step 1:
PROBLEM_LIST="/mnt/hum01-home01/d34195bc/av_datasets/${DATA_TYPE}/${CORPUS}/agg_problem_list.txt"

# ---- Select the Nth line from each list (N = SLURM_ARRAY_TASK_ID) ----
# Using awk with a variable for safety; exit after printing to avoid scanning whole file.
PROBLEM=$(awk -v n="$SLURM_ARRAY_TASK_ID" 'NR==n {print; exit}' "$PROBLEM_LIST")

KNOWN_LOC="/mnt/hum01-home01/d34195bc/av_datasets/${DATA_TYPE}/${CORPUS}/known_raw.jsonl"
UNKNOWN_LOC="/mnt/hum01-home01/d34195bc/av_datasets/${DATA_TYPE}/${CORPUS}/unknown_raw.jsonl"
METADATA_LOC="/mnt/hum01-home01/d34195bc/av_datasets/${DATA_TYPE}/metadata.rds"

for WEIGHT in linear power exp; do
    SAVE_LOC="/mnt/hum01-home01/d34195bc/av_datasets_experiments/weighted_ngram_tracing_concat/${DATA_TYPE}/${CORPUS}/${MODEL}/${WEIGHT}"
    COMPLETED_LOC="/mnt/hum01-home01/d34195bc/av_datasets_experiments/weighted_ngram_tracing_concat/${DATA_TYPE}/${CORPUS}/${MODEL}/${WEIGHT}"

    python -u /mnt/hum01-home01/d34195bc/projects/n-gram-av-methods/scripts/weighted_ngram_tracing_concat.py \
        --known_loc "${KNOWN_LOC}" \
        --unknown_loc "${UNKNOWN_LOC}" \
        --metadata_loc "${METADATA_LOC}" \
        --model_loc "${MODEL_LOC}" \
        --save_loc "${SAVE_LOC}" \
        --completed_loc "${COMPLETED_LOC}" \
        --corpus "${CORPUS}" \
        --data_type "${DATA_TYPE}" \
        --problem "${PROBLEM}" \
        --weight "${WEIGHT}" \
        --base "${BASE}" \
        --alpha "${ALPHA}" \
        --lowercase
done
