set -e

# Load conda module
module load apps/binapps/anaconda3/2024.10

# Activate the environment
conda activate venv

BASE_LOC="/mnt/hum01-home01/d34195bc/av_datasets_experiments/weighted_ngram_tracing_concat_v2"
SCRIPT_LOC="/mnt/hum01-home01/d34195bc/projects/n-gram-av-methods/scripts/weighted_ngram_tracing_rough_linears.py"

MODEL_LIST=(
    "gpt2"
)

CORPORA=(
    "ACL"
    "All-the-news"
    "Amazon"
    "Enron Cleaned"
    "IMDB"
    "Koppel's Blogs"
    "Perverted Justice"
    "Reddit"
    "StackExchange"
    "The Apricity"
    "The Telegraph"
    "TripAdvisor"
    "Wiki"
    "Yelp"
)

DATA_TYPES=(
    "test"
    "training"
)

for DATA_TYPE in "${DATA_TYPES[@]}"; do
    for CORPUS in "${CORPORA[@]}"; do
        for MODEL in "${MODEL_LIST[@]}"; do

            DATA_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/linear"
            SAVE_LOC="${BASE_LOC}/${DATA_TYPE}/${CORPUS}/${MODEL}/linear_rough"

            if [ ! -d "${DATA_LOC}" ]; then
                echo "Skipping missing directory: ${DATA_LOC}"
                continue
            fi

            CMD=(
                python -u "${SCRIPT_LOC}"
                --input_dir "${DATA_LOC}"
                --save_dir "${SAVE_LOC}"
            )

            "${CMD[@]}"
        done
    done
done
