#!/bin/bash --login
set -euo pipefail

# ------------------------------------------------------------
# Expected environment variables from the generated jobscript:
#   DATA_TYPE
#   CORPUS
#   MODEL
#   MODEL_LOC
#   PROBLEM_LIST
#   DATASET_BASE
#   RESULTS_BASE
#   METADATA_LOC
#   PYTHON_SCRIPT
#   POWER_ALPHAS_STR
#   EXP_BASES_STR
# ------------------------------------------------------------

# Load conda module
module load apps/binapps/anaconda3/2024.10

# Activate environment
conda activate venv

echo "----------------------------------------"
echo "DATA_TYPE: ${DATA_TYPE}"
echo "CORPUS: ${CORPUS}"
echo "MODEL: ${MODEL}"
echo "MODEL_LOC: ${MODEL_LOC}"
echo "PROBLEM_LIST: ${PROBLEM_LIST}"
echo "SLURM_ARRAY_TASK_ID: ${SLURM_ARRAY_TASK_ID}"
echo "----------------------------------------"

if [[ ! -f "${PROBLEM_LIST}" ]]; then
    echo "Problem list not found: ${PROBLEM_LIST}"
    exit 1
fi

# Get the Nth non-empty line from the problem list
PROBLEM=$(awk -v n="${SLURM_ARRAY_TASK_ID}" 'NF {c++; if (c == n) {print; exit}}' "${PROBLEM_LIST}")

if [[ -z "${PROBLEM:-}" ]]; then
    echo "No problem found for SLURM_ARRAY_TASK_ID=${SLURM_ARRAY_TASK_ID}"
    exit 1
fi

KNOWN_LOC="${DATASET_BASE}/${DATA_TYPE}/${CORPUS}/known_raw.jsonl"
UNKNOWN_LOC="${DATASET_BASE}/${DATA_TYPE}/${CORPUS}/unknown_raw.jsonl"

run_one() {
    local WEIGHT="$1"
    local ALPHA="$2"
    local BASE="$3"
    local SAVE_LOC="$4"

    mkdir -p "${SAVE_LOC}"

    echo
    echo "Running:"
    echo "  problem = ${PROBLEM}"
    echo "  weight  = ${WEIGHT}"
    echo "  alpha   = ${ALPHA}"
    echo "  base    = ${BASE}"
    echo "  save    = ${SAVE_LOC}"

    python -u "${PYTHON_SCRIPT}" \
        --known_loc "${KNOWN_LOC}" \
        --unknown_loc "${UNKNOWN_LOC}" \
        --metadata_loc "${METADATA_LOC}" \
        --model_loc "${MODEL_LOC}" \
        --save_loc "${SAVE_LOC}" \
        --completed_loc "${SAVE_LOC}" \
        --corpus "${CORPUS}" \
        --data_type "${DATA_TYPE}" \
        --problem "${PROBLEM}" \
        --weight "${WEIGHT}" \
        --alpha "${ALPHA}" \
        --base "${BASE}" \
        --lowercase
}

# ------------------------------------------------------------
# none
# ------------------------------------------------------------
run_one \
    "none" \
    "1.0" \
    "2.0" \
    "${RESULTS_BASE}/${DATA_TYPE}/${CORPUS}/${MODEL}/none"

# ------------------------------------------------------------
# linear
# ------------------------------------------------------------
run_one \
    "linear" \
    "1.0" \
    "2.0" \
    "${RESULTS_BASE}/${DATA_TYPE}/${CORPUS}/${MODEL}/linear"

# ------------------------------------------------------------
# power (subfolder by alpha)
# ------------------------------------------------------------
IFS=' ' read -r -a POWER_ALPHAS <<< "${POWER_ALPHAS_STR}"

for ALPHA in "${POWER_ALPHAS[@]}"; do
    run_one \
        "power" \
        "${ALPHA}" \
        "2.0" \
        "${RESULTS_BASE}/${DATA_TYPE}/${CORPUS}/${MODEL}/power/alpha_${ALPHA}"
done

# ------------------------------------------------------------
# exp (subfolder by base)
# ------------------------------------------------------------
IFS=' ' read -r -a EXP_BASES <<< "${EXP_BASES_STR}"

for BASE in "${EXP_BASES[@]}"; do
    run_one \
        "exp" \
        "1.0" \
        "${BASE}" \
        "${RESULTS_BASE}/${DATA_TYPE}/${CORPUS}/${MODEL}/exp/base_${BASE}"
done

echo
echo "Finished successfully for problem: ${PROBLEM}"
