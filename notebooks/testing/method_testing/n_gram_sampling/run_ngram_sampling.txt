#!/bin/bash --login
# REFACTORED: one SLURM array task per problem from agg_problem_list.txt.

#SBATCH -n 1

set -euo pipefail

# Load conda module
module load apps/binapps/anaconda3/2024.10

# Activate the environment
conda activate venv

# ============================================================
# Required variables passed in from the master submitter
# ============================================================

: "${CORPUS:?Missing CORPUS}"
: "${DATA_TYPE:?Missing DATA_TYPE}"
: "${MODEL:?Missing MODEL}"

: "${DATA_ROOT:?Missing DATA_ROOT}"
: "${MODEL_ROOT:?Missing MODEL_ROOT}"
: "${EXPERIMENT_ROOT:?Missing EXPERIMENT_ROOT}"
: "${EXPERIMENT_NAME:?Missing EXPERIMENT_NAME}"
: "${PYTHON_SCRIPT:?Missing PYTHON_SCRIPT}"
: "${PROBLEM_LIST_NAME:?Missing PROBLEM_LIST_NAME}"
: "${COMPLETED_SUBDIR:?Missing COMPLETED_SUBDIR}"
: "${ERROR_SUBDIR:?Missing ERROR_SUBDIR}"

: "${N_SAMPLES:?Missing N_SAMPLES}"
: "${MAX_ATTEMPTS:?Missing MAX_ATTEMPTS}"
: "${SEED:?Missing SEED}"
: "${SCORE_COL:?Missing SCORE_COL}"

# Optional settings supplied by the master submitter
MIN_LEN="${MIN_LEN:-}"
MAX_LEN="${MAX_LEN:-}"
LOWERCASE="${LOWERCASE:-true}"
GREATEST_COMMON="${GREATEST_COMMON:-true}"
USE_BOS="${USE_BOS:-true}"
USE_MIN_TOKEN_SIZE="${USE_MIN_TOKEN_SIZE:-true}"
OVERWRITE="${OVERWRITE:-false}"

# ============================================================
# Build paths
# ============================================================

PROBLEM_LIST="${DATA_ROOT}/${DATA_TYPE}/${CORPUS}/${PROBLEM_LIST_NAME}"
KNOWN_LOC="${DATA_ROOT}/${DATA_TYPE}/${CORPUS}/known_raw.jsonl"
UNKNOWN_LOC="${DATA_ROOT}/${DATA_TYPE}/${CORPUS}/unknown_raw.jsonl"
METADATA_LOC="${DATA_ROOT}/${DATA_TYPE}/metadata.rds"
MODEL_LOC="${MODEL_ROOT}/${MODEL}"

OUTPUT_ROOT="${EXPERIMENT_ROOT}/${EXPERIMENT_NAME}/${DATA_TYPE}/${CORPUS}/${MODEL}"
SAVE_LOC="${OUTPUT_ROOT}/${COMPLETED_SUBDIR}"
ERROR_SAVE_LOC="${OUTPUT_ROOT}/${ERROR_SUBDIR}"

mkdir -p "${SAVE_LOC}" "${ERROR_SAVE_LOC}"

# ============================================================
# Select the problem for this array task
# Count only non-empty problem-list lines
# ============================================================

PROBLEM="$(awk -v n="${SLURM_ARRAY_TASK_ID}" 'NF {c++} c==n {print; exit}' "${PROBLEM_LIST}")"

if [[ -z "${PROBLEM:-}" ]]; then
    echo "ERROR: No problem found for array task ${SLURM_ARRAY_TASK_ID}"
    echo "Problem list: ${PROBLEM_LIST}"
    exit 1
fi

# ============================================================
# Display job configuration
# ============================================================

echo "============================================================"
echo "EXPERIMENT         : ${EXPERIMENT_NAME}"
echo "DATA_TYPE          : ${DATA_TYPE}"
echo "CORPUS             : ${CORPUS}"
echo "MODEL              : ${MODEL}"
echo "ARRAY_TASK_ID      : ${SLURM_ARRAY_TASK_ID}"
echo "PROBLEM            : ${PROBLEM}"
echo "PROBLEM_LIST       : ${PROBLEM_LIST}"
echo "SAVE_LOC           : ${SAVE_LOC}"
echo "ERROR_SAVE_LOC     : ${ERROR_SAVE_LOC}"
echo "N_SAMPLES          : ${N_SAMPLES}"
echo "MAX_ATTEMPTS       : ${MAX_ATTEMPTS}"
echo "SEED               : ${SEED}"
echo "SCORE_COL          : ${SCORE_COL}"
echo "MIN_LEN            : ${MIN_LEN:-None}"
echo "MAX_LEN            : ${MAX_LEN:-None}"
echo "LOWERCASE          : ${LOWERCASE}"
echo "GREATEST_COMMON    : ${GREATEST_COMMON}"
echo "USE_BOS            : ${USE_BOS}"
echo "USE_MIN_TOKEN_SIZE : ${USE_MIN_TOKEN_SIZE}"
echo "OVERWRITE          : ${OVERWRITE}"
echo "============================================================"

# ============================================================
# Build Python arguments
# ============================================================

PYTHON_ARGS=(
    --known_loc "${KNOWN_LOC}"
    --unknown_loc "${UNKNOWN_LOC}"
    --metadata_loc "${METADATA_LOC}"
    --model_loc "${MODEL_LOC}"
    --save_loc "${SAVE_LOC}"
    --error_save_loc "${ERROR_SAVE_LOC}"
    --corpus "${CORPUS}"
    --data_type "${DATA_TYPE}"
    --problem "${PROBLEM}"
    --n_samples "${N_SAMPLES}"
    --max_attempts "${MAX_ATTEMPTS}"
    --seed "${SEED}"
    --score_col "${SCORE_COL}"
)

if [[ -n "${MIN_LEN}" ]]; then
    PYTHON_ARGS+=(--min_len "${MIN_LEN}")
fi

if [[ -n "${MAX_LEN}" ]]; then
    PYTHON_ARGS+=(--max_len "${MAX_LEN}")
fi

if [[ "${LOWERCASE}" == "true" ]]; then
    PYTHON_ARGS+=(--lowercase)
else
    PYTHON_ARGS+=(--no-lowercase)
fi

if [[ "${GREATEST_COMMON}" == "true" ]]; then
    PYTHON_ARGS+=(--greatest_common)
else
    PYTHON_ARGS+=(--no-greatest_common)
fi

if [[ "${USE_BOS}" == "true" ]]; then
    PYTHON_ARGS+=(--use_bos)
else
    PYTHON_ARGS+=(--no-use_bos)
fi

if [[ "${USE_MIN_TOKEN_SIZE}" == "true" ]]; then
    PYTHON_ARGS+=(--use_min_token_size)
else
    PYTHON_ARGS+=(--no-use_min_token_size)
fi

if [[ "${OVERWRITE}" == "true" ]]; then
    PYTHON_ARGS+=(--overwrite)
fi

# ============================================================
# Run one problem
# ============================================================

python -u "${PYTHON_SCRIPT}" "${PYTHON_ARGS[@]}"

echo "Finished array task successfully."
