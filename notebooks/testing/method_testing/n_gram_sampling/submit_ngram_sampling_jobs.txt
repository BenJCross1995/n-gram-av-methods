#!/bin/bash --login
# REFACTORED: submits MODEL -> CORPUS -> DATA_TYPE arrays, with test before training.

#SBATCH -J submit_ngram_sampling
#SBATCH -p himem
#SBATCH -t 7-0
#SBATCH -n 1
#SBATCH --mem=64G

set -euo pipefail

# Load conda module
module load apps/binapps/anaconda3/2024.10

# Activate the environment
conda activate venv

# ============================================================
# Central shared settings
# ============================================================

PROJECT_ROOT="/mnt/hum01-home01/d34195bc/projects/n-gram-av-methods"

DATA_ROOT="/mnt/hum01-home01/d34195bc/av_datasets"
MODEL_ROOT="/mnt/hum01-home01/d34195bc/models"
EXPERIMENT_ROOT="/mnt/hum01-home01/d34195bc/av_datasets_experiments"

EXPERIMENT_NAME="ngram_sampling"

# This filename is unchanged, but it now processes one --problem per call.
PYTHON_SCRIPT="${PROJECT_ROOT}/notebooks/testing/method_testing/n_gram_sampling/score_ngram_sampling_corpus.py"

JOB_SCRIPT="${PROJECT_ROOT}/notebooks/testing/method_testing/n_gram_sampling/run_ngram_sampling.txt"
PROBLEM_LIST_NAME="agg_problem_list.txt"

COMPLETED_SUBDIR="completed"
ERROR_SUBDIR="errors"

# Logs
LOG_DIR="${PROJECT_ROOT}/slurm_logs/${EXPERIMENT_NAME}"
mkdir -p "${LOG_DIR}"

# Shared SLURM settings for child array jobs
CHILD_PARTITION="himem"
CHILD_TIME="7-0"
CHILD_NTASKS="1"
CHILD_MEM="64G"

# When true, the training array for a model/corpus starts only after the
# corresponding test array has finished. afterany allows training to run even
# when individual test tasks record errors.
CHAIN_DATA_TYPES=true

# ============================================================
# Experiment settings
# ============================================================

N_SAMPLES=10
MAX_ATTEMPTS=1000
SEED=42
SCORE_COL="sum_log_probs"

# Leave blank to avoid filtering before scoring.
MIN_LEN=""
MAX_LEN=""

LOWERCASE=true
GREATEST_COMMON=true
USE_BOS=true
USE_MIN_TOKEN_SIZE=true

# Existing files in completed/ are skipped unless this is true.
OVERWRITE=false

# ============================================================
# Lists to edit
# ============================================================

MODEL_LIST=(
    "gpt2"
)

CORPORA=(
    "ACL"
    "All-the-news"
    "Amazon"
    "Enron Cleaned"
    "IMDB"
    "Koppel's Blogs"
    "Perverted Justice"
    "Reddit"
    "StackExchange"
    "The Apricity"
    "The Telegraph"
    "TripAdvisor"
    "Wiki"
    "Yelp"
)

CORPORA=(
    "Enron Cleaned"
    "StackExchange"
    "Perverted Justice"
    "Wiki"
)

# This order is respected inside each model/corpus pair.
DATA_TYPES=(
    "test"
    "training"
)

# ============================================================
# Helpers
# ============================================================

slugify() {
    local x="$1"
    x="${x,,}"        # lowercase
    x="${x// /_}"     # spaces -> underscores
    x="${x//\'/}"     # remove apostrophes
    x="${x//,/}"      # remove commas
    x="${x//\//_}"    # slashes -> underscores
    x="${x//(/}"      # remove (
    x="${x//)/}"      # remove )
    printf '%s\n' "$x"
}

count_nonempty_lines() {
    local file="$1"
    awk 'NF {c++} END {print c+0}' "$file"
}

# ============================================================
# Validate shared files before submitting anything
# ============================================================

if [[ ! -f "${PYTHON_SCRIPT}" ]]; then
    echo "ERROR: Python script does not exist:"
    echo "  ${PYTHON_SCRIPT}"
    exit 1
fi

if [[ ! -f "${JOB_SCRIPT}" ]]; then
    echo "ERROR: Child jobscript does not exist:"
    echo "  ${JOB_SCRIPT}"
    exit 1
fi

# ============================================================
# Submit arrays in MODEL -> CORPUS -> DATA_TYPE order
# ============================================================

SUBMITTED_COUNT=0
SKIPPED_COUNT=0

for MODEL in "${MODEL_LIST[@]}"; do
    MODEL_LOC="${MODEL_ROOT}/${MODEL}"

    if [[ ! -e "${MODEL_LOC}" ]]; then
        echo "Skipping missing model:"
        echo "  ${MODEL_LOC}"
        echo
        SKIPPED_COUNT=$((SKIPPED_COUNT + 1))
        continue
    fi

    for CORPUS in "${CORPORA[@]}"; do
        # Reset dependency for each new model/corpus pair.
        PREVIOUS_DATA_TYPE_JOB_ID=""

        for DATA_TYPE in "${DATA_TYPES[@]}"; do
            METADATA_LOC="${DATA_ROOT}/${DATA_TYPE}/metadata.rds"
            KNOWN_LOC="${DATA_ROOT}/${DATA_TYPE}/${CORPUS}/known_raw.jsonl"
            UNKNOWN_LOC="${DATA_ROOT}/${DATA_TYPE}/${CORPUS}/unknown_raw.jsonl"
            PROBLEM_LIST="${DATA_ROOT}/${DATA_TYPE}/${CORPUS}/${PROBLEM_LIST_NAME}"

            MISSING_INPUT=false

            for REQUIRED_FILE in \
                "${METADATA_LOC}" \
                "${KNOWN_LOC}" \
                "${UNKNOWN_LOC}" \
                "${PROBLEM_LIST}"; do
                if [[ ! -f "${REQUIRED_FILE}" ]]; then
                    echo "Skipping missing input:"
                    echo "  ${REQUIRED_FILE}"
                    echo
                    MISSING_INPUT=true
                    break
                fi
            done

            if [[ "${MISSING_INPUT}" == "true" ]]; then
                SKIPPED_COUNT=$((SKIPPED_COUNT + 1))
                continue
            fi

            NUM_TASKS="$(count_nonempty_lines "${PROBLEM_LIST}")"

            if [[ "${NUM_TASKS}" -eq 0 ]]; then
                echo "Skipping empty problem list:"
                echo "  ${PROBLEM_LIST}"
                echo
                SKIPPED_COUNT=$((SKIPPED_COUNT + 1))
                continue
            fi

            JOB_NAME="ngram_sampling_$(slugify "${MODEL}")_$(slugify "${CORPUS}")_$(slugify "${DATA_TYPE}")"

            echo "------------------------------------------------------------"
            echo "Submitting:"
            echo "  MODEL              = ${MODEL}"
            echo "  CORPUS             = ${CORPUS}"
            echo "  DATA_TYPE          = ${DATA_TYPE}"
            echo "  PROBLEM_LIST       = ${PROBLEM_LIST}"
            echo "  NUM_TASKS          = ${NUM_TASKS}"
            echo "  JOB_NAME           = ${JOB_NAME}"
            echo "  COMPLETED_SUBDIR   = ${COMPLETED_SUBDIR}"
            echo "  ERROR_SUBDIR       = ${ERROR_SUBDIR}"

            SBATCH_ARGS=(
                --parsable
                --job-name "${JOB_NAME}"
                --array "1-${NUM_TASKS}"
                -p "${CHILD_PARTITION}"
                -t "${CHILD_TIME}"
                -n "${CHILD_NTASKS}"
                --mem "${CHILD_MEM}"
                -o "${LOG_DIR}/%x_%A_%a.out"
                -e "${LOG_DIR}/%x_%A_%a.err"
            )

            if [[ \
                "${CHAIN_DATA_TYPES}" == "true" \
                && -n "${PREVIOUS_DATA_TYPE_JOB_ID}" \
            ]]; then
                SBATCH_ARGS+=(
                    --dependency="afterany:${PREVIOUS_DATA_TYPE_JOB_ID}"
                )
                echo "  DEPENDS ON         = ${PREVIOUS_DATA_TYPE_JOB_ID}"
            fi

            JOB_ID_RAW="$(
                CORPUS="${CORPUS}" \
                DATA_TYPE="${DATA_TYPE}" \
                MODEL="${MODEL}" \
                DATA_ROOT="${DATA_ROOT}" \
                MODEL_ROOT="${MODEL_ROOT}" \
                EXPERIMENT_ROOT="${EXPERIMENT_ROOT}" \
                EXPERIMENT_NAME="${EXPERIMENT_NAME}" \
                PYTHON_SCRIPT="${PYTHON_SCRIPT}" \
                PROBLEM_LIST_NAME="${PROBLEM_LIST_NAME}" \
                COMPLETED_SUBDIR="${COMPLETED_SUBDIR}" \
                ERROR_SUBDIR="${ERROR_SUBDIR}" \
                N_SAMPLES="${N_SAMPLES}" \
                MAX_ATTEMPTS="${MAX_ATTEMPTS}" \
                SEED="${SEED}" \
                SCORE_COL="${SCORE_COL}" \
                MIN_LEN="${MIN_LEN}" \
                MAX_LEN="${MAX_LEN}" \
                LOWERCASE="${LOWERCASE}" \
                GREATEST_COMMON="${GREATEST_COMMON}" \
                USE_BOS="${USE_BOS}" \
                USE_MIN_TOKEN_SIZE="${USE_MIN_TOKEN_SIZE}" \
                OVERWRITE="${OVERWRITE}" \
                sbatch "${SBATCH_ARGS[@]}" "${JOB_SCRIPT}"
            )"

            # --parsable may append a cluster name after a semicolon.
            JOB_ID="${JOB_ID_RAW%%;*}"

            echo "  SUBMITTED ID       = ${JOB_ID_RAW}"
            echo

            PREVIOUS_DATA_TYPE_JOB_ID="${JOB_ID}"
            SUBMITTED_COUNT=$((SUBMITTED_COUNT + 1))
        done
    done
done

echo "All submissions complete."
echo "Submitted array jobs: ${SUBMITTED_COUNT}"
echo "Skipped combinations: ${SKIPPED_COUNT}"
